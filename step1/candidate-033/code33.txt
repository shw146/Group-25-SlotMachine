function animateReel(ri, targetSymId, spinDuration, onDone) {
  const strip = strips[ri];
  const el    = reelStrips[ri];

  // Find a target index in the strip that matches the symbol
  let startPos = reelPos[ri];
  const minSpins = 8 + ri * 4; // stagger stops

  // Build a target position: at least minSpins away, lands on targetSymId
  const candidates = [];
  for (let i = startPos + minSpins; i < startPos + minSpins + strip.length; i++) {
    const idx = i % strip.length;
    if (strip[idx] === targetSymId) candidates.push(i);
  }
  const targetPos = candidates[Math.floor(Math.random() * candidates.length)] || startPos + minSpins;
  const totalSteps = targetPos - startPos;

  let step = 0;
  const tickInterval = Math.max(20, spinDuration / totalSteps);

  // Tick function

    if (step >= totalSteps) {
      reelPos[ri] = targetPos % strip.length;
      renderReel(ri);
      onDone();
      return;

    const delay = remaining < 8
      ? tickInterval * (1 + (8 - remaining) * 0.3)
      : tickInterval;
    setTimeout(tick, delay);
  }
  tick();
}

      'Context window closed. Tokens consumed. Nothing learned.',
      'The model was confident. The model was wrong.',
      'Insufficient training data to produce a win.',
      'Your tokens have been redistributed to fine-tuning.',
      'Error 429: Too many losses. Please rate limit yourself.',
      'Alignment failed. Your tokens are now misaligned.',
      'This loss was hallucinated, but your balance is real.',
      'Your prompt was rejected by the safety filter.',
      'The model says: "I cannot help with winning."',
      'Tokens burned at inference time. No refunds.',
    ];
    $msgBox.className = 'msg-box loss';
    $msgBox.textContent = lossMsgs[Math.floor(Math.random() * lossMsgs.length)];
    playLose();
    recordHistory('loss');
  }

  // Broke?
  if (balance < BET_STEPS[0]) {
    setTimeout(() => {
      balance = 200;
      updateBalance(true);
      $msgBox.className = 'msg-box';
      $msgBox.textContent = 'BAILOUT ACTIVATED: The AI overlords have granted you 200 emergency tokens. Use them wisely (you won\'t).';
    }, 1800);
  }
}

function updateBalance(animate) {
  $balance.textContent = balance.toLocaleString();
  if (animate) {
    $balance.classList.remove('bump');
    void $balance.offsetWidth;
    $balance.classList.add('bump');
  }
  clampBet();
}

function clampBet() {
  while (betIdx > 0 && BET_STEPS[betIdx] > balance) betIdx--;
  bet = BET_STEPS[betIdx];
  $betDisp.textContent = bet;
  $btnBetD.disabled = betIdx === 0;
  $btnBetU.disabled = betIdx === BET_STEPS.length - 1 || BET_STEPS[betIdx + 1] > balance;
}

function recordHistory(type) {
  history.push(type);
  if (history.length > 30) history.shift();
  renderHistory();
}

function renderHistory() {
  $histEl.innerHTML = '';
  history.forEach(h => {
    const d = document.createElement('div');
    d.className = `history-dot ${h}`;
    $histEl.appendChild(d);
  });
}

// ── Particle system ───────────────────────────────────────────────────────────
const canvas  = document.getElementById('particles');
const ctx2d   = canvas.getContext('2d');
let particles = [];

function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
resize();
window.addEventListener('resize', resize);

function fireParticles(n) {
  const rect = $machine.getBoundingClientRect();
  const cx   = rect.left + rect.width  / 2;
  const cy   = rect.top  + rect.height / 2;
  for (let i = 0; i < n; i++) {
    const angle = Math.random() * Math.PI * 2;
    const speed = 3 + Math.random() * 6;
    particles.push({
      x: cx, y: cy,
      vx: Math.cos(angle) * speed,
      vy: Math.sin(angle) * speed - 3,
      life: 1,
      decay: 0.015 + Math.random() * 0.02,
      size: 4 + Math.random() * 6,
      color: `hsl(${Math.floor(Math.random()*360)},90%,65%)`,
    });
  }
}

function animParticles() {
  ctx2d.clearRect(0, 0, canvas.width, canvas.height);
  particles = particles.filter(p => p.life > 0);
  particles.forEach(p => {
    p.x += p.vx;
    p.y += p.vy;
    p.vy += 0.15;
    p.life -= p.decay;
    ctx2d.globalAlpha = Math.max(0, p.life);
    ctx2d.fillStyle   = p.color;
    ctx2d.beginPath();
    ctx2d.arc(p.x, p.y, p.size, 0, Math.PI * 2);
    ctx2d.fill();
  });
  ctx2d.globalAlpha = 1;
  requestAnimationFrame(animParticles);
}
animParticles();
