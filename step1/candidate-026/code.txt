      598 function ctx() {

      601   return actx;

      604 function tone(freq, type, dur, vol=0.25, t=0) {
      605   try {
      606     const c=ctx(), o=c.createOscillator(), g=c.createGain();
      607     o.connect(g); g.connect(c.destination);
      608     o.type=type; o.frequency.value=freq;
      609     const now=c.currentTime+t;
      610     g.gain.setValueAtTime(0,now);
      611     g.gain.linearRampToValueAtTime(vol,now+0.01);
      612     g.gain.exponentialRampToValueAtTime(0.001,now+dur);
      613     o.start(now); o.stop(now+dur+0.05);
      614   } catch(e){}
      615 }
      616
      617 function playSpin() {
      618   try {
      619     const c=ctx(), o=c.createOscillator(), g=c.createGain();
      620     o.connect(g); g.connect(c.destination);
      621     o.type='sawtooth';
      622     const now=c.currentTime;
      623     o.frequency.setValueAtTime(180,now);
      624     o.frequency.linearRampToValueAtTime(420,now+0.4);
      625     o.frequency.linearRampToValueAtTime(120,now+0.8);
      626     g.gain.setValueAtTime(0.04,now);
      627     g.gain.linearRampToValueAtTime(0.001,now+0.8);
      628     o.start(now); o.stop(now+0.9);
      629   } catch(e){}
      630 }

      657 function spinReel(idx, finalSym, spinCount, duration) {
      658   return new Promise(resolve => {
      659     const reel  = document.getElementById('reel'+idx);
      660     const strip = document.getElementById('strip'+idx);
      661
      662     // Build strip: [spinCount random] + [final] + [2 random]
      663     const syms = [];
      664     for(let i=0;i<spinCount;i++) syms.push(rndSym());
      665     const resultIdx = syms.length;
      666     syms.push(finalSym);
      667     syms.push(rndSym());
      668     syms.push(rndSym());
      669
      670     // Populate DOM
      671     strip.style.transition = 'none';
      672     strip.innerHTML = '';
      673     syms.forEach((e,i) => {
      674       const d = document.createElement('div');
      675       d.className = 'sym' + (i===resultIdx?' center':'');
      676       d.textContent = e;
      677       strip.appendChild(d);
      678     });
      679
      680     // Start position: symbol[1] centered  →  translateY = 0
      681     strip.style.transform = 'translateY(0)';
      682
      683     reel.classList.add('spinning');
      684     reel.classList.remove('landed');
      685
      686     // Force reflow then kick off animation
      687     void strip.offsetHeight;
      688
      689     // End position: symbol[resultIdx] centered
      690     const endY = -(resultIdx - 1) * SH;
      691
      692     strip.style.transition = `transform ${duration}ms cubic-bezier(0.05, 0.0, 0.25, 1.0)`;
      693     strip.style.transform   = `translateY(${endY}px)`;
      694
      695     setTimeout(() => {
      696       reel.classList.remove('spinning');
      697       reel.classList.add('landed');
      698       playClick();
      699       setTimeout(() => {
      700         reel.classList.remove('landed');
      701         resolve();
      702       }, 420);
      703     }, duration);
      704   });
      705 }

      708 // WIN EVALUATION
      709 // ────────────────────────────────────────────
      710 function evaluate(res) {
      711   const [a,b,c] = res;
      712   const key = a+b+c;
      713
      714   if(TRIPLES[key]) return TRIPLES[key];
      715
      716   if(a===b && b===c) return { m:15, lv:'medium', msg: pick(WIN_MSGS) };
      717
      718   // 2× robot bonus
      719   const robotCount = res.filter(x=>x==='🤖').length;
      720   if(robotCount>=2) return { m:10, lv:'small', msg:'🤖🤖 TWO ROBOTS plotting to reward you. Trust the process.' };
      721
      722   if(a===b || b===c || a===c) return { m:2, lv:'small', msg: pick(WIN_MSGS) };
      723
      724   return null;
      725 }

      799 // MAIN SPIN

      801 async function spin() {
      802   if(spinning) return;
      803
      804   if(tokens < bet) {
      805     setMsg('⚠ INSUFFICIENT TOKENS. The model cannot generate funds from thin air. (Usually.)', 'lose');
      806     if(tokens <= 0) {
      807       setTimeout(()=>{
      808         tokens = 100; refreshBalance();
      809         setMsg('♻ BAILOUT COMPLETE. 100 FREE TOKENS courtesy of the Federal Reserve AI Fund.', 'win');
      810         litWinLine(true);
      811         setTimeout(()=>litWinLine(false), 1500);
      812       }, 1800);
      813     }
      814     return;
      815   }
      816
      817   spinning = true;
      818   const btn = document.getElementById('spinBtn');
      819   btn.disabled = true;
      820   btn.textContent = '⟳  RUNNING INFERENCE…';
      821
      822   litWinLine(false);
      823
      824   // Deduct bet
      825   const prevTokens = tokens;
      826   tokens -= bet;
      827   stats.spins++;
      828   stats.net -= bet;
      829   flashBal('flash-loss');
      830   refreshBalance();
      831   updateStats();
      832
      833   setMsg('> SAMPLING FROM DISTRIBUTION…');
      834   playSpin();
      835
      836   // Determine results (weighted random)
      837   const results = [rndSym(), rndSym(), rndSym()];
      838   const winData  = evaluate(results);
      839
      840   // Staggered reel durations
      841   const configs = [
      842     { count:22, dur:1800 },
      843     { count:28, dur:2400 },
      844     { count:34, dur:3000 },
      845   ];
      846
      847   await Promise.all(results.map((sym,i) =>
      848     spinReel(i, sym, configs[i].count, configs[i].dur)
      849   ));
      850
      851   // Resolve outcome
      852   if(winData) {
      853     const payout = Math.floor(bet * winData.m);
      854     tokens += payout;
      855     stats.wins++;
      856     stats.net += payout;
      857     if(payout > stats.big) stats.big = payout;
      858
      859     document.getElementById('lastWinVal').textContent = '+'+payout;
      860     document.getElementById('lastWinVal').style.color = 'var(--win)';
      861
      862     flashBal('flash-gain');
      863     refreshBalance();
      864     litWinLine(true);
      865     setTimeout(()=>litWinLine(false), 1800);
      866
      867     setMsg(`+${payout} TOKENS   ${winData.msg}`, winData.lv==='jackpot' ? 'jackpot' : 'win');
      868     playWin(winData.lv);
      869
      870     const pCount = winData.lv==='jackpot' ? 60
      871                  : winData.lv==='big'     ? 28
      872                  : winData.lv==='medium'  ? 14
      873                  :                          6;
      874     particles(pCount, winData.lv);
      875
      876   } else {
      877     document.getElementById('lastWinVal').textContent = '—';
      878     document.getElementById('lastWinVal').style.color = 'var(--dim)';
      879
      880     setMsg(pick(LOSE_MSGS), 'lose');
      881     playLose();
      882
      883     const m = document.getElementById('machine');
      884     m.classList.remove('shake');
      885     void m.offsetWidth;
      886     m.classList.add('shake');
      887     setTimeout(()=>m.classList.remove('shake'), 400);
      888   }
      889
      890   updateStats();
      891
      892   // Bankruptcy check
      893   if(tokens <= 0) {
      894     tokens = 0; refreshBalance();
      895     setTimeout(()=>{
      896       setMsg('💀 BANKRUPTCY DECLARED. Liquidity crisis detected. Initiating emergency stimulus…', 'lose');
      897       setTimeout(()=>{
      898         tokens = 75; refreshBalance();
      899         setMsg('🏦 STIMULUS APPROVED: +75 TOKENS. "Too big to fail" applies to slot machines too.', 'win');
      900         particles(20, 'medium');
      901         litWinLine(true);
      902         setTimeout(()=>litWinLine(false), 1500);
      903       }, 2200);
      904     }, 600);
      905   }
      906
      907   spinning = false;
      908   btn.disabled = false;
      909   btn.textContent = '▶  GENERATE TOKENS  ◀';
      910 }
